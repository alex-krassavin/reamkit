import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="From the spec" icon="open-book">
    Built directly from **ECMA-376** (OOXML) and **ISO 32000 / 19005** (PDF / PDF/A) — not a wrapper
    around LibreOffice, headless Office, or any commercial SDK. Existing implementations are read as
    a reference, never a dependency.
  </Card>
  <Card title="Browser-first" icon="laptop">
    Pure TypeScript working on `Uint8Array` in and `Uint8Array` out — zero `node:*` imports on the
    conversion path. The same code runs unchanged in the browser, Node.js, serverless and edge
    runtimes. You supply the fonts (or let it fetch open substitutes).
  </Card>
  <Card title="Seven formats in" icon="document">
    Office Open XML — **`.docx`** (text, styles, tables, lists, multi-column sections,
    headers/footers with real page numbers, footnotes, drawings, tracked changes), **`.xlsx`**
    (grids, number formats, the print model, conditional formatting, charts) and **`.pptx`** (slides,
    placeholders, shapes, tables, charts). Plus **`.pdf`** and the legacy binary **`.doc` / `.xls` /
    `.ppt`** (97–2003). DrawingML shapes, charts and OfficeMath throughout.
  </Card>
  <Card title="Real typography" icon="seti:font">
    Type0 + CIDFontType2 embedding with subsetting, Knuth–Plass line breaking, Liang hyphenation,
    OpenType ligatures and kerning, BiDi (UAX #9) and Arabic cursive joining.
  </Card>
  <Card title="Compliant PDF" icon="approve-check">
    PDF/A-1, -2 and -3 (levels a / b / u) and **PDF/UA-1** — all formally **veraPDF-validated** —
    plus tagged PDF (logical structure, alt text), digital signatures (PKCS#7 detached, RSA / ECDSA,
    PAdES, RFC 3161 timestamps) and AES-256 encryption.
  </Card>
  <Card title="One parse, any target" icon="rocket">
    `Ream.parse(bytes)` reads any of the seven formats once into a format-neutral interlayer;
    `convert('pdf' | 'svg' | 'html' | 'docx' | 'xlsx')` renders from it — so a `.doc` becomes a
    `.docx`, a `.pptx` becomes a PDF, a PDF becomes flowed HTML. Formats plug in as reader/writer
    adapters.
  </Card>
</CardGrid>
